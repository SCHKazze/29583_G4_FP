//CRIOLLO JAHIR EJC1
#include <stdio.h>
#include <string.h>

int main(void)
{
    char frase[200];
    int letras = 0, espacios = 0;

    printf("Ingrese una frase: ");
    fgets(frase, sizeof(frase), stdin);

    frase[strcspn(frase, "\n")] = '\0';

    for (int i = 0; frase[i] != '\0'; i++)
    {
        if (frase[i] == ' ')
            espacios++;
        else
            letras++;
    }

    printf("Letras: %d\n", letras);
    printf("Espacios: %d\n", espacios);

    return 0;
}

#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main(void)
{
    char texto[200], limpio[200];
    int j = 0;

    printf("Ingrese una cadena: ");
    fgets(texto, sizeof(texto), stdin);
    texto[strcspn(texto, "\n")] = '\0';

    for (int i = 0; texto[i] != '\0'; i++)
    {
        if (texto[i] != ' ')
            limpio[j++] = (char)tolower((unsigned char)texto[i]);
    }
    limpio[j] = '\0';

    int esPal = 1;
    for (int i = 0, k = j - 1; i < k; i++, k--)
    {
        if (limpio[i] != limpio[k])
        {
            esPal = 0;
            break;
        }
    }

    if (esPal)
        printf("Es un palindromo\n");
    else
        printf("No es un palindromo\n");

    return 0;
}

//CRIOLLO JAHIR EJC6
#include <stdio.h>
#include <string.h>

void invertir(char s[])
{
    int i = 0;
    int j = (int)strlen(s) - 1;

    while (i < j)
    {
        char aux = s[i];
        s[i] = s[j];
        s[j] = aux;
        i++;
        j--;
    }
}

int main(void)
{
    char texto[100], copia[100];

    printf("Ingrese una cadena: ");
    fgets(texto, sizeof(texto), stdin);
    texto[strcspn(texto, "\n")] = '\0';

    strcpy(copia, texto);
    invertir(copia);

    printf("Original: %s\n", texto);
    printf("Invertida: %s\n", copia);

    return 0;
}

//CRIOLLO JAHIR EJC7
#include <stdio.h>
#include <ctype.h>

int contarVocales(const char *s)
{
    int cont = 0;

    for (int i = 0; s[i] != '\0'; i++)
    {
        char c = (char)tolower((unsigned char)s[i]);
        if (c=='a' || c=='e' || c=='i' || c=='o' || c=='u')
            cont++;
    }
    return cont;
}

int main(void)
{
    char texto[100];

    printf("Ingrese una cadena: ");
    fgets(texto, sizeof(texto), stdin);

    printf("Cantidad de vocales: %d\n", contarVocales(texto));

    return 0;
}
